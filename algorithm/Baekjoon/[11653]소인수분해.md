# 코드

```cpp
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main() {

	int num;
	cin >> num;

	for (int i = 2; i <= sqrt(num); i++)
	{
		if (num % i == 0)
		{
			printf("%d\n", i);
			num /= i,i = 1;
		}
	}
	if (num != 1)
		printf("%d\n", num);

	return 0;
}
```
