# 코드

```cpp
#include <iostream>
#include <math.h>
#include <vector>
using namespace std;

int main() {
	int x ,y ,res = 0,front = -1;
	cin >> x >> y;
	vector<bool> vec = vector<bool>(y + 1,false);

	for (int i = 2; i <= y; i++)
		if (vec[i] == false)
			for (int j = i * 2; j <= y ; j += i)
				vec[j] = true;

	
	for (int i = x; i < y + 1; i++)
	{
		if (i > 1 && vec[i] == 0)
		{
			if (front == -1)
				front = i;
			res += i;
		}
	}
	if (front == -1)
		cout << -1 << endl;
	else
		cout << res <<endl<< front << endl;
	return 0;
}
```
