# 코드

```cpp

#include<iostream>
#include<vector>
using namespace std;

int main()
{
	int n;
	bool isIn = false;
	scanf("%d",&n);
	vector<int> veclist = vector<int>();
	for (int i = 0; i < n; i++)
	{
		int num;
		scanf("%d", &num);
		veclist.push_back(num);

		for (int j = i - 1; j > -1; j--)
		{
			if (veclist[j] > veclist[i])
			{
				printf("%d ", j + 1);
				isIn = true;
				break;
			}
		}
		if (isIn == false)
			printf("0 ");
		isIn = false;
	}

	return 0;
}
```
