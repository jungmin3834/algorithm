# 코드

```cpp
#include <iostream>
#include <math.h>
#include <vector>
using namespace std;

int main() {
	int num,res = 0;
	cin >> num;
	vector<bool> vec = vector<bool>(1001,false);
	for (int i = 2; i <= 1000; i++)
		if (vec[i] == false)
			for (int j = i * 2; j <= 1000; j += i)
				vec[j] = true;

	for (int i = 0; i < num; i++)
	{
		int n; cin >> n;
		res += (n == 0 || n == 1) ? 0 : vec[n] == 0 ? 1 : 0;
	}
	cout << res << endl;
	return 0;
}
```
