# 코드

```cpp

#include <iostream>
#include <vector>
using namespace std;

int main() {

	int x, y;
	scanf("%d %d", &x , &y);
	vector<bool> vec = vector<bool>(y + 1, false);

	for (int i = 2; i<=y; i++)
		if (vec[i] == false)
			for (int j = i * 2; j <= y; j += i)
				vec[j] = true;

	x = x > 1 ? x : 2;
	for (int i = x; i <= y; i++)
		if (vec[i] == false)
			printf("%d\n", i);


	return 0;
}
```
