# 코드

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
	int n ;
	cin >> n;
	vector<vector<int>> vec = vector<vector<int>>(n * 2 - 1,vector<int>(n + 1,0));
	for(int i = 0;i < n;i++)
	{
		for(int j = 1;j < i + 2;j++)
		{
				int num;
				cin >> num;
				vec[i][j] = num +  (i != 0 ? vec[i][j] += max(vec[i - 1][j - 1] , vec[i - 1][j]) : 0);
		}
	}
	
	for(int i = n;i > 0 ;i--)
	{
		for(int j = 1;j < i;j++)
		{
				int num;
				cin >> num;
				vec[(n * 2 ) - i][j] = num + max(vec[(n * 2) - i - 1][j], vec[(n * 2 ) - i - 1][j + 1]);
		}
	}
	
	cout << vec.back()[1] <<endl;
	

	return 0;
}
```
