# 코드

```cpp
#include <iostream>
#include <queue>
#include <string>
using namespace std;
int main() {

	queue<int> q = queue<int>();

	int n,num;
	char ch;

	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> ch;
		if (ch == 'e' || ch == 'E')
		{
			cin >> num;
			if (q.size() == 10)
				cout <<("overflow")<< endl;	
			else
				q.push(num);
		}
		else if (ch == 'd' || ch == 'D')
		{
			if (q.size() == 0)
				cout << ("underflow") <<endl;	
			else
				q.pop();
		}
	}
	
	for (int i = q.size(); i > 0; i--)
	{
		cout << q.front()  << (i == 1 ? "": " ");
		q.pop();
	}


	return 0;
}
```
