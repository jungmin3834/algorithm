1389. Create Target Array in the Given Order

# 코드 리뷰 
```cpp
class Solution {
public:
    vector<int> createTargetArray(vector<int>& nums, vector<int>& index) {
        vector<int> res = vector<int>();
	    for (int i = 0; i < nums.size(); i++)
		    res.insert(res.begin() + index[i], nums[i]);
	    return res;
    }
};
```
