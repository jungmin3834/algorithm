1018. Binary Prefix Divisible By 5

# 문제 링크

# 코드

```cpp
class Solution {
public:
    vector<bool> prefixesDivBy5(vector<int>& A) {
        vector<bool> res = vector<bool>(A.size(), false);
	    int sum = 0;
	    A[0] == 0 ? res[0] = true : sum = 1;
    	for (int i = 1; i < A.size(); i++)
    	{
            sum = ((sum << 1) + A[i]) % 5;
		    res[i] =  sum == 0 ? true : false;
	    }
	    return res;
    }
};
```
