1380. Lucky Numbers in a Matrix

# 코드

```cpp
class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        
    vector<int> res = vector<int>();
	int m = matrix.size();
	int n = matrix[0].size();
	for (int i = 0; i < m; i++)
	{
		int idx = -1;
		int row = INT_MAX;

		for (int j = 0; j < n; j++)
			if (row > matrix[i][j])
				row = matrix[i][j], idx = j;

		res.push_back(row);
		for (int j = 0; j < m; j++)
		{
			if (matrix[j][idx] > row)
			{
				res.pop_back();
				break;
			}
		}
	}
	return res;
        
    }
};
```
