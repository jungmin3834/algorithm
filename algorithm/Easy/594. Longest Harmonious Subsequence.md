# 코드

```cpp
class Solution {
public:
    int findLHS(vector<int>& nums) {
    int res = 0;
	map<int, int> maplist = map<int, int>();
	for (int i = 0; i<nums.size(); i++)
		maplist[nums[i]]++;

	int size = 0;
	for (auto i = maplist.begin(); i != maplist.end(); i++)
	{
        if(i->second == 0)
            continue;
		res = max(res , maplist[i->first + 1] == 0 ? 0 : i->second + maplist[i->first + 1]);
	}

	return res;
    }
};
```
