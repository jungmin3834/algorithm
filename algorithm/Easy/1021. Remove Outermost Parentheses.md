# 코드


```java

class Solution {
    public String removeOuterParentheses(String S) {
        int n =0;
		String res = "";
		for(int i =0;i<S.length();i++) {
			n = S.charAt(i) == '(' ? n + 1 : n - 1;
			res += (n > 1 || (n == 1 && S.charAt(i) == ')')) ? S.charAt(i) : "";
		}
		return res;
    }
}
```

```cpp
class Solution {
public:
    string removeOuterParentheses(string S) {
        string res = "";
        int count = 0;
        bool start = false;
        for(int i = 0 ;i < S.size();i++)
        {  
            S[i] =='(' ? count += 1 : count -= 1;
            if(count > 0 && start != false)
                res.push_back(S[i]);
            if(start == false && count == 1)
                start = true;
            else if(start == true && count == 0)
                start = false;   
        }
        return res;
    }
};
```

Stack Solution
```cpp
class Solution {
public:
    string removeOuterParentheses(string S) {
        stack<int> q = stack<int>();
        for(int i = S.size() -1 ;i > -1;i--)
        {
            if(q.empty() == false && S[q.top()] == ')' && S[i] == '(')
            {
                int idx = q.top(); q.pop();
                if(q.empty() == true)
                    S.erase(S.begin() + idx) ,S.erase(S.begin() + i);
            }
            else
                q.push(i);
        }
        return S;
    }
};
```
