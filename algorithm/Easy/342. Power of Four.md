# 코드

```cpp
class Solution {
public:
    bool isPowerOfFour(int n) {
        long result = 1;
        int i = 0;
        for(i =0;i<31;i++){
            result =pow(4,i);
            if(result == n)
                return true;
            else if(result > n)
                return false;
        }
        return result + (i == 31 ? 0 : 1) == n ? true : false;
    }
};
```
