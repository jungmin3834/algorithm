# 코드

```cpp
class Solution {
public:
    int bitwiseComplement(int N, int res =0, int count = 0) {
        while(true)
        {
            N % 2 == 0 ? (res += 1 << count) : (res += 0),  N >>= 1 , count++;
            if(N == 0)
                return res;
        }
    }
};
```
