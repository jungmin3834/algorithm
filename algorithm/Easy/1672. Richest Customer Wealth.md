1672. Richest Customer Wealth

# 코드

```cpp
class Solution {
public:
    int maximumWealth(vector<vector<int>>& accounts) {
        int res = 0;
        for(int i  =0;i<accounts.size();i++){
            int temp = 0;
            for(int j =0;j<accounts[i].size();j++){
                
                temp += accounts[i][j];
            }
            res = temp > res ? temp : res;
        }
        return res;
    }
};
```
