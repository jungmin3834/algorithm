```cpp
class Solution {
public:
    int numIdenticalPairs(vector<int>& nums,vector<int> vec =vector<int>(101,0),int res = 0) {
        for(int i =0;i<nums.size();i++)
            res += vec[nums[i]]++;
        return res;
    }
};
```

```cpp
class Solution {
public:
    int numIdenticalPairs(vector<int>& nums) {
        vector<int> vec =vector<int>(101,0);
        int res = 0;
        int size = nums.size();
        for(int i =0;i<size;i++){
            res += vec[nums[i]]++;
        }
        return res;
    }
};
```
