# 코드

```cpp
#include <string>
#include <vector>

using namespace std;

int dps(vector<int> numbers, int target, int idx)
{
    if (idx == numbers.size())
		return target == 0 ? 1 : 0;
	return dps(numbers, target + numbers[idx], idx + 1) + dps(numbers, target - numbers[idx], idx + 1);
}

int solution(vector<int> numbers, int target) {
	return dps(numbers, target, 0);
}
```
