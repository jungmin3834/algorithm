```sql
SELECT A1.ANIMAL_ID ,A1.NAME FROM ANIMAL_OUTS A1 
WHERE NOT EXISTS (SELECT * FROM ANIMAL_INS A2 WHERE A1.ANIMAL_ID=A2.ANIMAL_ID) 
ORDER BY A1.ANIMAL_ID ASC                                                                   
```
