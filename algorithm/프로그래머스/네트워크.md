# 코드

```cpp
#include <string>
#include <vector>
using namespace std;

bool dp[201];
void helper(int idx , vector<vector<int>> computers)
{
	dp[idx] = true;
	for (int i = 0; i < computers.size(); i++)
		if (!dp[i] && computers[idx][i] == 1)
			helper(i, computers);
}

int solution(int n, vector<vector<int>> computers) {
	int res = 0;
	for (int i = 0; i < n; i++)
		if (!dp[i])
			helper(i, computers) , res++;
	return res;
}
```
