713. Subarray Product Less Than K

# 코드

```cpp

class Solution {
public:
int numSubarrayProductLessThanK(vector<int>& nums, int k) {
	int res = 0;
	int start = 0;
    int size = nums.size();
	long long sum = 1;
	for (int i = 0; i < size; i++) {

        sum *= nums[i];
        while (sum >= k && i > start)
		{
            res += i - start;
            sum /= nums[start++];
		}
            
		if (nums[i] >= k){
            sum = 1;
            start = i + 1;
        }
        
        
	}

    for(int i = start ;i<size;i++)
        res += size - i;
    
    
	return res;
}
};
```
