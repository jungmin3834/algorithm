150. Evaluate Reverse Polish Notation

# 코드

```cpp

class Solution {
public:
    int evalRPN(vector<string>& tokens) {
	stack<int> s = stack<int>();
	for (int i = 0; i < tokens.size(); i++)
	{
		if (tokens[i].back() > 47)
			s.push(stoi(tokens[i]));
		else
		{
			int a = s.top(); s.pop();
			int b = s.top(); s.pop();
			if (tokens[i][0] == '*')
				s.push(b * a);
			else if (tokens[i][0] == '-')
				s.push(b - a);
			else if (tokens[i][0] == '+')
				s.push(b + a);
			else if (tokens[i][0] == '/')
				s.push(b / a);
		}
	}
	return s.top();
}

};
```
