```cpp
class Solution {
public:
    vector<string> findRepeatedDnaSequences(string s) {
        
        vector<string> res = vector<string>();
        map<string, int> maplist = map<string,int>();
        
        string temp = "";
        for(int i =0;i<s.size();i++){
            temp.push_back(s[i]);
            if(temp.size() == 10){
                cout << temp <<endl;
                if(++maplist[temp] == 2)
                    res.push_back(temp);
                temp.erase(temp.begin());
            }
          
        }
        return res;
    }
};

```
