# 코드

```cpp
class Solution {
public:
    int minimumTotal(vector<vector<int>>& triangle) {
        for (int i = triangle.size() - 2; i > -1; i--)
		    for (int j = triangle[i].size() - 1; j > -1; j--)
			    triangle[i][j] += min(triangle[i + 1][j], triangle[i + 1][1 + j]);
	    return triangle.front().back();
    }
};
```
