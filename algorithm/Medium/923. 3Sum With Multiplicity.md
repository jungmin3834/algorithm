923. 3Sum With Multiplicity
  

# 문제 링크

# 코드

```cpp

class Solution {
public:
int threeSumMulti(vector<int>& A, int target) {
    int mod = 1000000007, len = A.size();
	long long res = 0;
	vector<int> m = vector<int>(100 + 1, 0);
	for (int i = 0; i <len; i++)
		m[A[i]]++;

	for (int i = 0; i < len; i++)
	{
		m[A[i]]--;
		vector<int> temp = m;
        int sum = target- A[i];
		for (int j = i + 1; j < len; j++)
		{
			temp[A[j]]--;
            if(sum - A[j] >= 0 && sum - A[j] <= 100)
                res = (res + temp[sum - A[j]]) % mod;   
		}
	}
    
	return res;
}
};

```
