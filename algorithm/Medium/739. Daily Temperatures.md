739. Daily Temperatures
  
  
  
# 문제 링크

# 코드

```cpp
class Solution {
public:
   vector<int> dailyTemperatures(vector<int>& T) {
        stack<int> s = stack<int>();
        for(int i =0;i<T.size();i++)
        {
			while (s.size() > 0 && T[s.top()] < T[i])
			{
				T[s.top()] = i - s.top();
				s.pop();
			}
			s.push(i);
        }
		
		for (int i = s.size(); i > 0; i--)
			T[s.top()] = 0 ,s.pop();
		
        return T;
 }
};
```
