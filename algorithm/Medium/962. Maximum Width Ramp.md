962. Maximum Width Ramp

  

# 문제 링크

# 코드

```cpp

class Solution {
public:
    int maxWidthRamp(vector<int>& A) {
    int len = A.size();
	int res = 0;
	vector<int> vec = vector<int>(1,0);
	for (int i = 0; i < len; i++)
	{
		if (A[vec.back()] > A[i])
			vec.push_back(i);
		else
		{
			for (int j = vec.size() - 1; j > -1; j--)
			{
				if (A[vec[j]] > A[i])
					break;
				res = max(i - vec[j], res);
			}
		}
	}
	return res;
    }
};

```
