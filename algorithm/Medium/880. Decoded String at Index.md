# 코드 
```cpp

class Solution {
public:
    string decodeAtIndex(string S, int K, long long len = 0) {
       	for (int i = 0; i < S.size(); i++)
	    {
		    if (isalpha(S[i]))
            {
		    	if (++len == K)
			    	return S.substr(i, 1);
            }
		    else
		    {
		    	if (len * (S[i] - '0') >= K)
		    		return decodeAtIndex(S.substr(0, i), K % len == 0 ? len : K % len);
		    	len *= (S[i] - '0');
		    }
    	}
        
	    return "";
    }
};
```
