1249. Minimum Remove to Make Valid Parentheses

# 코드

```cpp

class Solution {
public:
    string minRemoveToMakeValid(string s) {
        
   	stack<int> st = stack<int>();
	for (int i = 0; i < s.size(); i++)
	{
		if (s[i] == '(')
			st.push(i);
		else if (s[i] == ')')
		{
			if (st.size() == 0 || s[st.top()] == ')')
				st.push(i);
			else
				st.pop();
		}
	}
	
	while (st.size() != 0)
		s.erase(s.begin() + st.top(), st.pop();
	
	
	return s;
    }
};
```
