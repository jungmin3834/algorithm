1014. Best Sightseeing Pair
  
# 문제 링크

# 코드

```cpp

class Solution {
public:
    int maxScoreSightseeingPair(vector<int>& A) {
        int prev = A[0], res = 0;
        
        for(int i = 1;i<A.size();i++){
            res = max(res , (A[i] - i) + prev);
            prev = max(prev , A[i] + i);
        }
        
        return res;
    }
};

```

Time Out Solution 

```cpp  
  
class Solution {
public:
    int maxScoreSightseeingPair(vector<int>& A) {
        int res = 0;
        for(int i =0;i<A.size() - 1;i++){
            for(int j =i + 1;j<A.size();j++){
                res = max(res , A[i] + A[j] + i - j);
            }
        }
        return res;
    }
};
```  
