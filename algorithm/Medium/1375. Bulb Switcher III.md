# 코드 

```cpp
class Solution {
public:
    int numTimesAllBlue(vector<int>& light) {
    vector<bool> vec = vector<bool>(light.size() + 1,false);
	int res = 0, count = 0,start= 1;
	for (int i = 0; i < light.size(); i++)
	{
		vec[light[i]] = true;
		if (light[i] == start)
		{
			for (int j = start; j < vec.size(); j++)
			{
				if (vec[j] == true)
					count++, start++;
				else
					break;
			}
			if (count == i + 1)
				res++;
		}
	}

	return res;
}
};
```
