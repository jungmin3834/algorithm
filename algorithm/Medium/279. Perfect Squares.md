279. Perfect Squares
  

# 문제 링크    
https://leetcode.com/problems/perfect-squares/

# 코드

```cpp

class Solution {
public:
int numSquares(int n) {
	vector<int> dp = vector<int>(n + 1, n + 1);
	dp[0] = 0;
	for (int i = 1; i <= n; i++)
	{
		int sqrtNum = i * i;
		if (sqrtNum > n)
			break;
        dp[sqrtNum] = 1;
		for (int j = sqrtNum + 1; j < dp.size(); j++)
			 dp[j] = min(dp[j - sqrtNum] + 1, dp[j]);	
	}
	return dp.back();
}
};

```
