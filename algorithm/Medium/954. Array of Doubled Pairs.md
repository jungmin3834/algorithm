# 코드

```cpp
class Solution {
public:
bool canReorderDoubled(vector<int>& A) {

	sort(A.begin(), A.end());
	int res = 0;
	map<int, int> maplist = map<int, int>();

	for (int i = 0; i < A.size(); i++)
		maplist[A[i]]++;

	for (int i = 0; i < A.size(); i++)
	{
		if (maplist[A[i]] != 0 && maplist[A[i] * 2] != 0)
		{
			--maplist[A[i]];
			--maplist[A[i] * 2];
			res++;
		}
	}

	return res * 2 == A.size();
}
};
```
