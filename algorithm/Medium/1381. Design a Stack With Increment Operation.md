```cpp
class CustomStack {
public:
  	vector<int> vec;
	vector<int> plus;
	int msize = 0, idx = 0;
	CustomStack(int maxSize) {
		msize = maxSize;
		vec = vector<int>(msize,0);
		plus = vector<int>(maxSize + 1, 0);
	}

	void push(int x) {
		if (msize == idx)
			return;
        plus[idx] = 0;
		vec[idx++] = x;
	}

	int pop() {
       	if (idx < 1)
			return -1;
		--idx;
		int res = vec[idx] + plus[idx];
        if(idx != 0)
            plus[idx - 1] += plus[idx];
		return  res;
	}

	void increment(int k, int val) {
        if(idx == 0)
            return;
		k = (k >= idx ?  idx : k);
		plus[k - 1] += val;
	}
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */

```

```cpp
class CustomStack {
public:
   	vector<int> vec = vector<int>();
	map<int, int> maplist = map<int, int>();
	int msize = 0,size = 0;
    
	CustomStack(int maxSize) {
		msize = maxSize;
	}

	void push(int x) {
		if (size == msize)
			return;
		size++;
        vec.push_back(x);
	}

	int pop() {

		if (size == 0)
			return -1;
        --size;
		int plusNum = 0, res = vec.back();
		vec.pop_back();

		if (maplist.rbegin() != maplist.rend() && size < maplist.rbegin()->first)
		{
            plusNum += maplist.rbegin()->second;
			maplist.erase(maplist.rbegin()->first);
			if(size != 0)
				maplist[size] += plusNum;
        }
		
		return res + plusNum;
	}

	void increment(int k, int val) {
		k = (k >= size ? size : k);
        if(size != 0)
		    maplist[k] += val;
	}
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
```
