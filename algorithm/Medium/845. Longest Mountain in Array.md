845. Longest Mountain in Array

# 코드

```cpp

class Solution {
public:
    int longestMountain(vector<int>& A) {
    int res = 0;
    int len = A.size();
	for (int i = 1; i < len - 1; i++)
	{
		if (A[i - 1] < A[i] && A[i] > A[i + 1])
		{
			int count = 1;
			for (int j = i - 1; j > -1; j--)
			{
				if (A[j] < A[j + 1])
					count++;
				else
					break;
			}

			for (int j = i + 1; j < A.size();j++)
			{
				if (A[j - 1] > A[j])
					count++;
				else
				{
					i = j - 1;
					break;
				}
			}
			res = max(res, count);
		}
	}

	return res;
    }
};

```
