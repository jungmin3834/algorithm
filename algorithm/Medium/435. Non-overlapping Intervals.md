# 코드

```cpp

class Solution {
public:
    int eraseOverlapIntervals(vector<vector<int>>& intervals) {
        sort(intervals.begin(), intervals.end());
	    int res = 0;
    	int x = INT_MAX;
	    int y = INT_MIN;
	    for (int i = 0; i < intervals.size(); i++)
	    {   
		    if (intervals[i][0] >= y)
            {
                x = intervals[i][0], y = intervals[i][1];
                continue;
            }
			x = min(intervals[i][0] , x) , y = min(y,intervals[i][1]);
            res++;
	    }
	    return res;
    }
};

```
