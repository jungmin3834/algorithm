967. Numbers With Same Consecutive Differences
  

# 문제 링크

# 코드

```cpp

class Solution {
public:
    vector<int> res = vector<int>();
    void helper(int n, int k,int cur,int temp){
        if(n == 0){
            res.push_back(temp);
            return;
        }
        if(cur - k >= 0 && k != 0)
            helper(n - 1 ,k , cur - k,(temp + cur - k) * (n != 1 ? 10 : 1));
        if(cur + k < 10)
            helper(n - 1 ,k , cur + k,(temp + cur + k) * (n != 1 ? 10 : 1));
         
    }
    
    vector<int> numsSameConsecDiff(int n, int k) {
        for(int i =1;i<10;i++)
            helper(n - 1,k,i,i * 10);
        return res;
    }
};

```
