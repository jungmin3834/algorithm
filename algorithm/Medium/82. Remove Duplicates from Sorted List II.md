# 코드

```cpp

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    
    ListNode* helper(ListNode *head){
        
        int pivot = head->val;
        if(head->next == NULL || head->val != head->next->val)
            return head;
        
        while(head)
        {
            head =  head->next;
            if(head != NULL && head->val != pivot)
                return helper(head);
        }
        return NULL;
    }
    
    
    ListNode* deleteDuplicates(ListNode* head) {
        ListNode* temp = head;
        ListNode* prev = NULL;
        while(temp)
        {
            if(prev == NULL)
                temp = head = helper(temp);
            else{
                prev->next = helper(temp);
            }
            prev = temp;
            if(temp != NULL)
                temp = temp->next;
        }
        
        return head;
    }
};
```
