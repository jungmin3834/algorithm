134. Gas Station

# 코드

```cpp
class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
	int size = gas.size();
	int res = -1;
    int station = -1;
	gas.insert(gas.end(), gas.begin(), gas.end());
	cost.insert(cost.end(), cost.begin(), cost.end());
	for (int i = 0; i < gas.size() / 2; i++) {
		int full = 0;
		for (int j = i; j < i + size; j++) {
			full += gas[j] - cost[j];
			if (full < 0)
				break;
		}
        if(station < full){
            res = i;
            station = full;
        }
		
	}

	return res == -1 ? -1 : res;
}

};
```
