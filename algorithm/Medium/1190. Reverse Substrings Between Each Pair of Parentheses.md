# 코드

```cpp

class Solution {
public:
    string reverseParentheses(string s) {
    string res = "";
    stack<int> pstack = stack<int>();
	for (int i = 0; i < s.size(); i++)
	{
		if (s[i] == '(')
			pstack.push(i);
		else if(s[i] == ')')
		{
			for (int j = pstack.top(), z = 0; j < i - (i - pstack.top()) / 2; j++, z++)
				swap(s[j], s[i - z]);		
			pstack.pop();
		}
	}
        
	for (int i = 0; i < s.size(); i++)
		if(s[i] != '(' && s[i] != ')')
			res.push_back(s[i]);

	return res;
    }
};
```
